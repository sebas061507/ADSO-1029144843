# CALCULAR AREA DEL RECTANGULO DADOS SUS LADOS
def calcular_area_del_rectangulo(lado1, lado2):
    return lado1 * lado2
    
# Ingresa un valor numerico
def ingresar_valor(mensaje):
    while True:
        try:
            x = float(input(mensaje))
            break
        except:
            print("caracter invalido")
            
    return x

lado1 = ingresar_valor("Ingresa la longitud del primer lado: ")

lado2 = ingresar_valor("Ingresa la longitud del segundo lado: ")

area = calcular_area_del_rectangulo(lado1,lado2)


print(f"El area del rectangulo es: {area}")
